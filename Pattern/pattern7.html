Pattern 7

n=int(input("Enter the value of n : "))
for i in range(n):
 for j in range(n-i-1):
  print(" ",end="")
 for j in range(2*i+1):
  print("*",end="")
 for j in range(n-i-1):
  print(" ",end="")
 print()
